assignment_expression ::=  target_list "=" assigned_expression
assigned_expression ::= starred_expression | yield_expression | assignment_expression
assignment_stmt ::= assignment_expression

augmented_assignment_expression ::=  augtarget augop aug_assigned
aug_assigned ::= expression_list | yield_expression
augmented_assignment_stmt ::= augtarget augop (expression_list | yield_expression) 
target ::= identifier
  | parenth_target_list
  | bracket_target_list
  | attributeref
  | subscription
  | slicing
  | starred_target

parenth_target_list ::= "(" void_target_list ")"
target_starred ::= "*" target
expression_starred ::= "*" expression
something_starred ::= "*" something

import_expr ::=  import_module | from_relative_module_import | from_module_import
from_module_import ::= "from" module "import" "*"
from_relative_module_import ::= "from" relative_module "import" identifier_named_list
import_module ::= "import" module ["as" name] ( "," module ["as" name] )*
import ::= import_module | from_relative_module_import | from_module_import
module ::= identifier_dotted_list
module_void ::= identifier_void_list
module_named_list ::= module_named "," ... "," module_named
module_named ::= module "as" name
identifier_named_list ::= identifier_named ( "," identifier_named )*
identifier_named ::= identifier "as" name

identifier_list ::= identifier ("," identifier)*
