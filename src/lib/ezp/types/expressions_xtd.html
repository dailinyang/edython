yield_expression ::=  yield_expression_list | yield_expression_from
yield_expression_list ::=  "yield" expression_list_or_void
yield_expression_from ::=  "yield from" expression

positional_arguments ::=  positional_argument ("," positional_argument)*
positional_argument ::= expression | starred_expression
starred_expression ::= "*" expression
starred_and_keywords ::=  starred_and_keyword ("," starred_and_keyword)*
starred_and_keyword ::=  starred_expression | keyword_item
keywords_arguments ::=  keywords_argument ("," keywords_argument)*
keywords_argument ::= keyword_item | double_starred_expression
double_starred_expression ::= "**" expression

u_expr ::=  power | u_expr_concrete
u_expr_concrete ::= "-" u_expr | "+" u_expr | "~" u_expr
m_expr ::=  u_expr | m_expr_concrete
m_expr_concrete ::= m_expr "*" u_expr | m_expr "@" m_expr | m_expr "//" u_expr| m_expr "/" u_expr | m_expr "%" u_expr
a_expr ::=  m_expr | a_expr_concrete
a_expr_concrete ::= a_expr "+" m_expr | a_expr "-" m_expr
shift_expr ::=  a_expr | shift_expr_concrete
shift_expr_concrete ::= shift_expr ( "<<" | ">>" ) a_expr
and_expr ::=  shift_expr | and_expr_concrete
and_expr_concrete ::= and_expr "&" shift_expr
xor_expr ::=  and_expr | xor_expr_concrete
xor_expr_concrete ::= xor_expr "^" and_expr
or_expr  ::=  xor_expr | or_expr_concrete
or_expr_concrete ::= or_expr "|" xor_expr

comparison ::=  or_expr | comparison_list
comparison_list ::=  or_expr comp_operator or_expr ( comp_operator or_expr )*

or_test  ::=  and_test | or_test_concrete
or_test_concrete  ::= or_test "or" and_test
and_test ::=  not_test | and_test_concrete
and_test_concrete ::= and_test "and" not_test
not_test ::=  comparison | not_test_concrete
not_test_concrete ::= "not" not_test

conditional_expression ::=  or_test | conditional_expression_concrete
conditional_expression_concrete ::=  or_test "if" or_test "else" expression

starred_expression ::=  expression | starred_expression_concrete
starred_expression_concrete ::=  ( starred_item "," )* [starred_item]
starred_item ::=  expression | starred_or_expr
starred_or_expr ::= "*" or_expr
