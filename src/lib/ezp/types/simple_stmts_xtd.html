target ::= identifier
  | parenth_target_list
  | bracket_target_list
  | attributeref
  | subscription
  | slicing
  | starred_target
bracket_target_list ::= "[" void_target_list "]"
parenth_target_list ::= "(" void_target_list ")"
starred_target ::= starred
starred_item ::= starred
starred_expression ::= starred
starred ::= "*" something
