target ::= identifier
  | parenth_target_list
  | bracket_target_list
  | attributeref
  | subscription
  | slicing
  | starred_target
bracket_target_list ::= "[" void_target_list "]"
parenth_target_list ::= "(" void_target_list ")"
starred_target ::= "*" target
starred_expression ::= "*" expression
starred_concrete ::= "*" something
starred_target ::= starred_concrete starred_item ::= starred_concrete starred_expression ::= starred_concrete starred_concrete ::= "*" something
